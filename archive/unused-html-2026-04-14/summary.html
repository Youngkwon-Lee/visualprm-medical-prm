⑥ 파이프라인 자동화 코드 (핵심 로직)

import openai, re

SOLVE_PROMPT = (
    "Your task is to answer the question below.\n"
    "Give step by step reasoning before you answer, "
    "and when you're ready to answer, use the format 'Final answer: ..'\n\n"
    "Question:\n{question}"
)

def generate_solutions(question, n=4):
    """문제당 n개 풀이 생성"""
    return [
        openai.chat.completions.create(
            model="gpt-4o",
            messages=[{"role": "user", "content": SOLVE_PROMPT.format(question=question)}],
            temperature=0.7,
        ).choices[0].message.content
        for _ in range(n)
    ]

def split_steps(solution):
    """풀이 → step 리스트"""
    return [s.strip() for s in re.split(r'\n\n+', solution.strip()) if s.strip()]

def monte_carlo_rollout(question, steps_so_far, gt_answer, n=16):
    """step_i까지 왔을 때 n번 계속 풀어보고 score 계산"""
    context = "\n\n".join(steps_so_far)
    prompt = f"Question:\n{question}\n\nSolution so far:\n{context}\n\nContinue. End with 'Final answer: ..'"
    correct = sum(
        gt_answer.lower() in openai.chat.completions.create(
            model="gpt-4o",
            messages=[{"role": "user", "content": prompt}],
            temperature=0.7,
        ).choices[0].message.content.lower()
        for _ in range(n)
    )
    return {"score": correct/n, "num_mc_correct": correct, "num_mc_total": n, "label": "+" if correct > 0 else "-"}

def label_solution(question, solution, gt_answer):
    """풀이 전체 step에 Monte Carlo 라벨 부여"""
    steps = split_steps(solution)
    return {
        "question": question,
        "response": solution,
        "steps_with_score": [
            {"step": s, **monte_carlo_rollout(question, steps[:i+1], gt_answer)}
            for i, s in enumerate(steps)
        ],
        "num_mc_sequences": 16
    }
